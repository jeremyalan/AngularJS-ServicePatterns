Decorator

Example
Modify a service using decorator
angular.module('api')
    .decorator('$person', function ($q, $delegate) {
        var personCache = {};
        
        var fetch = $delegate.fetch;
        $delegate.fetch = function (id) {
            if (personCache[id]) {
                return $q.when(personCache[id]);
            }
            
            return fetch(id);
        };
        
        return $delegate;
    });
